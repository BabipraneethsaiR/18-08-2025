package ExcHand;

public class MultithreadJoin {
	public static void main(String[] args){
		numeric n = new numeric();
		alphabets a= new alphabets();
		n.start();
		try{
			n.join();

		}
		catch(Exception e){
			System.out.println("sry for Inconvinence ");
		}
		a.start();
	}
	

}

class numeric extends Thread{
	int a = 0;
	public void run(){
		try{
			for(int i=0;i<10;i++){
				a++;
				System.out.println("The Numenric:"+a);
				
			}
		}
		catch(Exception e){
			System.out.println(e);
		}
	}
	
}

class alphabets extends Thread{
	char c='A';
	public void run(){
		try{
			for(int i=0;i<=25;i++){
				
				System.out.println("Alpha Thread:"+c);
				c++;
			}
		}
		catch(Exception e){
			System.out.println(e);
		}
	}
	
}
