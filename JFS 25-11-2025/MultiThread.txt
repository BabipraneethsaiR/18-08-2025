package ExcHand;

public class MultiThread {
	public static void main(String[] args){
		num n=new num();
		alpha a = new alpha();
		n.start();
		a.start();
		
	//Getting name for thread
		System.out.println("Getting name for thread that system Alloted"+n.getName());
	
	// Getting Priority that system gives to Thread
		System.out.println("Priority for thread as per system"+n.getPriority());
	// Checking state of thread 
		System.out.println("state of thread:"+n.getState());
		
	// we can also set name for thread as User_defined
		n.setName("Babi");
	// checking Weather Thread is alive or not 
		if(n.isAlive()){
			System.out.println("Numeric is Alive");
			
		}
		else{
			System.out.println("Dead");
		}
	
		
	}

}
class num extends Thread{
	int a = 0;
	public void run(){
		try{
			for(int i=0;i<10;i++){
				a++;
				System.out.println("The Numenric:"+a);
				
			}
		}
		catch(Exception e){
			System.out.println(e);
		}
	}
	
}

class alpha extends Thread{
	char c='A';
	public void run(){
		try{
			for(int i=0;i<=25;i++){
				
				System.out.println("Alpha Thread:"+c);
				c++;
			}
		}
		catch(Exception e){
			System.out.println(e);
		}
	}
	
}
