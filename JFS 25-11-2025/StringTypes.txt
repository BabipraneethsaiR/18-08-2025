package ExcHand;

public class StringTypes {
	public static void main(String[] args){
		StringBuilder sb = new StringBuilder("Welcome");
		sb.append("Chennai");
		System.out.println("appendig:"+sb);
		
		sb.insert(7, "Metro");
		System.out.println("Inserting at String_index:"+sb);
		
		sb.replace(7,13,"Hyderabad");
		
		System.out.println("Replacing Based on index values:"+sb);
		
		sb.delete(7,30);
		System.out.println("Deleting portion of String using index:"+sb);
		
		sb.reverse();
		System.out.println(sb);
	
			
		
	}

}
