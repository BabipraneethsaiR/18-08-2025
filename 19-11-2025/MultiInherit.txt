package nov19;

public class MulInherit {
	public static void main(String[] args){
		SonCla a3= new SonCla();
		a3.SonBike();
		a3.OldBike();
		
	}

}

class GrandFatherCla {
	void OldBike(){
		System.out.println("RX100");	
	}
}

class FatherCla extends GrandFatherCla{
	void OldBike(){
		super.OldBike();
		System.out.println("Honda Shine");
	}
}

class SonCla extends FatherCla{
	void SonBike(){
		System.out.println("KTM");
	
	}
}
